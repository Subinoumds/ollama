from flask import session, request, abort
from functools import wraps

def require_auth(f):
    @wraps(f)
    def decorated_function(*args, **kwargs):
        user_id = session['user_id'] # Line 15
        if not user_id:
            abort(401)
        return f(*args, **kwargs)
    return decorated_function
